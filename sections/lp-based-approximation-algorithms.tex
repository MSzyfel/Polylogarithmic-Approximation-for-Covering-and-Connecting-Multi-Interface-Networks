\section{LP based approximation algorithms}

Below we show an $O\br{\log m}$ approximation algorithms for all four variants and $O\br{k}$-approximation for the covering problems. The idea of our ILP formulation and the algorithm is inspired by the rounding procedure known for set cover \cite{vazirani2001approximation} and related problems \cite{Angelidakis2018ShortestPQ}

In the following sections, we assume that for every $v\in V$ and $i\in[k]$, $c\br{i, v}\geq 1$. If this is not a case, i. e., there exist $v\in V$ and $i\in[k]$, such that $c\br{i, v}=0$, then we apply the following rounding: for each $v\in V$ and $i\in[k]$ we set $c'\br{v, u}=nk\cdot c'\br{v, u}$. This transformation does not influence the optimality of the solution. Moreover, for $c\br{v, u} = 0$ set $c'\br{v, u}=1$. It is easy to see that in this modified instance $\OPT\geq nk$ for any cost critertion. This is because for the original instance $\OPT\geq 1$, otherwise activate all interfaces. Since the additional cost incurred by pairs $v\in V$ and $i\in[k]$, such that $c\br{v, u} = 0$ is at most $nk$, we see that this transformation worsens the approximation ratio of our algorithm at most $2$ times.
\subsection{ILP formulation of covering multi-interface networks}

We start with the following ILP formulation of the min-sum covering assignment problem. Let $x_{i, v}$ be a binary variable that is $1$ if and only if the interface $i\in\lambda\br{v}$ is assigned to vertex $v\in V$. Then the min-sum covering assignment problem can be formulated as follows:
\begin{align*}
    \text{min} \quad & \sum_{v\in V}\sum_{i\in\lambda\br{v}}c\br{i, v}\cdot x_{i, v}\\
    \text{s.t.} \quad & \sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}} \geq 1 \quad \forall uv\in E\\
    & x_{i, v} \in \{0, 1\} \quad \forall v\in V, i\in\lambda\br{v}
\end{align*}
Similarly, the min-max covering assignment problem can be formulated as follows:
\begin{align*}
    \text{min} \quad & M\\
    \text{s.t.} \quad & \sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} \leq M \quad \forall v\in V\\
    & \sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}} \geq 1 \quad \forall uv\in E\\
    & x_{i, v} \in \{0, 1\} \quad \forall v\in V, i\in\lambda\br{v}
\end{align*}

Note that the above formulations are not linear due to the presence of the $\min$ function. However, by adding an additional variable $y_{i, uv}$ we can linearize the constraints of type $\sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}} \geq 1$ by replacing them with $\sum_{i\in\lambda\br{u}\cap\lambda\br{v}}y_{i, uv} \geq 1$ together with the constraints $y_{i, uv} \leq x_{i, u}$ and $y_{i, uv} \leq x_{i, v}$ for every $i\in\lambda\br{u}\cap\lambda\br{v}$ and $uv\in E$. Thus, we can solve the LP relaxation of each of the above formulations in polynomial time.

\subsection{Randomized $O\br{\log m}$-approximation algorithm for the covering problem}

\begin{algorithm}[H]
\caption{A relaxation-based $O(\log m)$-approximation algorithm.}
Solve the LP relaxation and obtain a fractional solution $\{x_{i, v}\}_{i\in\lambda(v), v\in V}$.

Pick independent uniformly random thresholds $t_i \in (0, 1)$, for each $i \in [k]$.

Set $\lambda_A(v) = \{i \in \lambda(v) \colon 3\ln m\cdot x_{i, v} \geq t_i\}$, for every $v \in V$.

\Return{$\{\lambda_A(v)\}_{v\in V}$.}
\end{algorithm}

\begin{theorem}
The above algorithm is an $O(\log m)$-approximation algorithm for both the min-sum and the min-max covering assignment problems which succeeds with high probability.
\end{theorem}
\begin{proof}
We have the following lemmas:
\begin{lemma}
    The algorithm returns a solution of cost at most $9\ln m \cdot \OPT$ with high probability.
\end{lemma}
\begin{proof}
By $X_{v, i}$ denote the random variable that is $1$ if $3\ln m\cdot x_{i, v} \geq t_i$ and $0$ otherwise. Then we can write $c\br{\lambda_A\br{v}} = \sum_{i\in\lambda\br{v}}X_{v, i}\cdot c\br{i, v}$. We have: $\mathbb{E}[X_{v, i}] = \Pr[i \in \lambda_A\br{v}] = 3\ln m\cdot x_{i, v}$. Thus, $\mathbb{E}[c\br{\lambda_A\br{v}}] = 3\ln m \cdot \sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v}$. By applying the Chernoff bound with $\delta = 2$ we get that 
\begin{align*}
    \Pr[c\br{\lambda_A\br{v}} \geq 3\cdot \mathbb{E}[c\br{\lambda_A\br{v}}]] & 
    \leq \br{\frac{e^2}{3^3}}^{\mathbb{E}[c\br{\lambda_A\br{v}}]}
    \leq \e{-\mathbb{E}[c\br{\lambda_A\br{v}}]}\leq \frac{1}{m^3}
\end{align*}
where the last inequality is by using the fact that, $\sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} \geq 1$ for all $v \in V$. If otherwise, then $v$ would have no neighbors which contradicts connectivity. By applying the union bound over all $n$ vertices we get that with probability at least $1-\frac{n}{m^3}\geq 1-\frac{1}{m}$ (assuming $m\geq 2$) for every $v\in V$, $c\br{\lambda_A\br{v}} \leq 9\ln m \cdot \sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v}$. Thus, depending on the cost function, with high probability we have $\COST_A \leq 9\ln m \cdot \sum_{v\in V}\sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} = 9\ln m \cdot \OPT$ or $\COST_W \leq 9\ln m \cdot \max_{v\in V}\sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} = 9\ln m \cdot \OPT$. 
\end{proof}

    \begin{lemma}
With high probability, the above algorithm returns a covering assignment.
    \end{lemma}
\begin{proof}
    Consider an edge $uv\in E$. We have:
\begin{align*}
    \Pr[\lambda_A\br{u}\cap\lambda_A\br{v}=\emptyset] & = \prod_{i\in\lambda\br{u}\cap\lambda\br{v}}\Pr[t_i > 3\ln m \cdot \min\brc{x_{i, u}, x_{i, v}}]\\
    & = \prod_{i\in\lambda\br{u}\cap\lambda\br{v}}\brc{1-3\ln m \cdot\min\brc{x_{i, u}, x_{i, v}}}\\&
    \leq \prod_{i\in\lambda\br{u}\cap\lambda\br{v}}\e{-3\ln m \cdot\min\brc{x_{i, u}, x_{i, v}}} 
    \\& = \e{-3\ln m \cdot\sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}}}\\
    & \leq \e{-3\ln m} = \frac{1}{m^3}
\end{align*}
By applying the union bound over all $m$ edges we get that the algorithm returns a covering assignment with probability at least $1-\frac{m}{m^3} = 1-\frac{1}{m^2}$.
\end{proof}
Combining the above lemmas gives the claim.
\end{proof}
\subsection{A $k$-approximation algorithm for the covering problem}

\begin{algorithm}[H]
\caption{A relaxation-based $k$-approximation algorithm.}
Solve the LP relaxation and obtain a fractional solution $\{x_{i, v}\}_{i\in\lambda(v), v\in V}$.

Set $\lambda_A(v) = \{i \in \lambda(v) \colon x_{i, v} \geq \frac{1}{k}\}$, for every $v \in V$.

\Return{$\{\lambda_A(v)\}_{v\in V}$.}
\end{algorithm}
\begin{theorem}
The above algorithm is a $k$-approximation algorithm for both the min-sum and the min-max covering assignment problems.
\end{theorem}
\begin{proof}
Consider an edge $uv\in E$. By averaging, there exists an interface $i\in\lambda\br{u}\cap\lambda\br{v}$ such that $\min\brc{x_{i, u}, x_{i, v}} \geq \frac{1}{k}$. Thus, $i\in\lambda_A\br{u}\cap\lambda_A\br{v}$ so $\lambda_A$ is a covering assignment. Moreover, since $x_{i, v} \geq \frac{1}{k}$ for every $i\in\lambda_A\br{v}$, we have that $c\br{\lambda_A\br{v}} = \sum_{i\in\lambda_A\br{v}}c\br{i, v} \leq k \cdot \sum_{i\in\lambda_A\br{v}}x_{i, v}\cdot c\br{i, v}$. Thus, depending on the cost function, we have $\COST_A \leq k \cdot \sum_{v\in V}\sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} = k \cdot \OPT$ or $\COST_W \leq k \cdot \max_{v\in V}\sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} = k \cdot \OPT$.
\end{proof}
\subsection{ILP formulation for the connection problem}
We use a similar LP to model the connection problem, however we use additional flow variables to encode the connectivity requirement. The $x_{i, v}$ variables are defined as before. For each edge $e$ we add an additional flow variable $y_{e}$ which is $1$ if and only if the edge $e$ is covered by the solution. Connectivity of the terminals $U$ is enforced via cut constraints: we require that for every cut $(S, V\setminus S)$ that separates at least one pair of terminals, the total $y$-weight of edges crossing the cut is at least $1$. Then we can formulate the min-sum connection problem as follows:
\begin{align*}
    \text{min} \quad & \sum_{v\in V}\sum_{i\in\lambda\br{v}}c\br{i, v}\cdot x_{i, v}\\
    \text{s.t.} \quad & \sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}} \geq y_{uv} \quad \forall uv\in E\\
    & y\br{\delta\br{S}}\geq 1 \quad \forall S\subset V\colon S\cap U\neq\emptyset,\, \br{V\setminus S}\cap U\neq\emptyset\\
    & x_{i, v}  \in \{0, 1\} \quad \forall v\in V, i\in\lambda\br{v}\\
    & y_{uv} \in \{0, 1\} \quad \forall uv\in E
\end{align*}

Similarly, the min-max connection problem can be formulated as follows:
\begin{align*}
    \text{min} \quad & M\\
    \text{s.t.} \quad & \sum_{i\in\lambda\br{v}}x_{i, v}\cdot c\br{i, v} \leq M \quad \forall v\in V\\
    & \sum_{i\in\lambda\br{u}\cap\lambda\br{v}}\min\brc{x_{i, u}, x_{i, v}} \geq y_{uv} \quad \forall uv\in E\\
    & y\br{\delta\br{S}}\geq 1 \quad \forall S\subset V\colon S\cap U\neq\emptyset,\, \br{V\setminus S}\cap U\neq\emptyset\\
    & x_{i, v}  \in \{0, 1\} \quad \forall v\in V, i\in\lambda\br{v}\\
    & y_{uv} \in \{0, 1\} \quad \forall uv\in E
\end{align*}
    
Observe that both of the above ILPs have exponential number of constraints, but we can solve their LP relaxations in polynomial time by using the ellipsoid method and a separation oracle for the cut constraints, which can be implemented using a max-flow algorithm.

\subsection{Randomized $O\br{\log^2m}$-approximation algorithm for the connection problem}

\begin{algorithm}[H]
\caption{A relaxation-based $O(\log^2 m)$-approximation algorithm.}
Solve the LP relaxation and obtain a fractional solution $\{x_{i, v}\}_{i\in\lambda(v), v\in V}, \{y_{uv}\}_{uv\in E}$.

For every edge $e\in E$, let $p_e = \min\{1, 3\ln m \cdot y_e\}$. Pick independent uniformly random variables $Y_e$ such that $Y_e = 1$ with probability $p_e$ and $Y_e = 0$ otherwise. Let $H$ be the subgraph of $G$ induced by the edges $e$ such that $Y_e = 1$.

Use the $O\br{\log m}$-approximation algorithm for the covering problem on $H$ to obtain a covering assignment $\lambda_A$ for $H$.

\Return{$\{\lambda_A(v)\}_{v\in V}$.}
\end{algorithm}
\begin{theorem}
    There exists an $O\br{\log^2 m}$-approximation algorithm for both the min-sum and the min-max connection assignment problems which succeeds with high probability.
\end{theorem}
\begin{proof}
    We show that the algorithm for the covering problem described in the previous section can be used to obtain an $O\br{\log^2 m}$-approximation algorithm for the connection problem. We start with the following lemma:
    \begin{lemma}
        With high probability, the subgraph $H$ connects all terminals in $U$.
    \end{lemma}
    \begin{proof}
        Fix any $S\subset V$ with $S\cap U\neq\emptyset$ and $\br{V\setminus S}\cap U\neq\emptyset$. Let $\ell\in\mathbb{N}$, such that $\ell\leq y\br{\delta\br{S}}<\ell+1$.
        We have:
        \begin{align*}
            \Pr[\spr{\delta_H\br{S}}=0] & = \prod_{e\in\delta\br{S}}\Pr[Y_e=0] = \prod_{e\in\delta\br{S}}(1-p_e) \\&\leq \prod_{e\in\delta\br{S}}\e{-p_e} \leq \e{-\sum_{e\in\delta(S)}\min\brc{1,\, 3\ln m\cdot y_e}}
            \\& \leq \e{-3\ell\ln m} = \frac{1}{m^{3\ell}}
        \end{align*}
        where the last inequality uses the LP constraint $y\br{\delta\br{S}}\geq 1$ for every $U$-separating cut, so $\sum_{e\in\delta(S)} y_e \geq \ell$.

        A cut in $G$ is called an $\alpha$-minimum cut if its value is less than $\alpha$ times the value of a minimum cut in $G$. 
        We now make a use of the following Karger's lemma \cite{MinimumCutsInNearLinearTime, RandomSamplingInCutFlowAndNetworkDesignProblems}:
        \begin{lemma}[Karger's lemma]
            The number of $\alpha$-minimum cuts in a graph is at most $n^{2\alpha}$.
        \end{lemma}
        Observe that by construction of the LP, every $U$-separating fractional cut has a value of at least $1$, so there are at most $n^{2\ell}$ $U$-separating cuts of value less than $\ell$. By applying the union bound over all $U$-separating cuts we get that:
\begin{align*}
    \Pr[\exists\, U\text{-sep. cut } S\colon \spr{\delta_H\br{S}}=0] & \leq \sum_{\ell=1}^{\infty}\sum_{\substack{S\subset V,\,S\cap U\neq\emptyset\\ (V\setminus S)\cap U\neq\emptyset,\; y\br{\delta\br{S}}<\ell+1}}\Pr[\spr{\delta_H\br{S}}=0] \\&\leq \sum_{\ell=1}^{\infty}n^{2\ell}\cdot \frac{1}{m^{3\ell}} = \sum_{\ell=1}^{\infty}\frac{1}{m^{\ell}} = \frac{1}{m-1}
\end{align*}
Thus, with high probability, no $U$-separating cut is empty in $H$, i.e., $H$ spans $U$.
\end{proof}

Now we want to show that by covering $H$ we can obtain good approximation for connecting $G$. We have the following lemma:
\begin{lemma}
    With high probability, the cost of covering $H$ is at most $9\ln m \cdot \OPT$.
\end{lemma}
\begin{proof}
    Let $\bar{\mathbf{x}}, \bar{\mathbf{y}}$ be the fractional solution such that every variable is scaled up by a factor of $3\ln m$. Clearly, this is a valid LP solution. Moreover, by the linearity of the LP objective function, we know that the cost of $\bar{\mathbf{x}}, \bar{\mathbf{y}}$ is at most $3\ln m \cdot \OPT$. Since for every edge $e\in E$, we have that $\Pr[Y_e=1] = \min\{1, 3\ln m \cdot y_e\}$, by linearity of expectations we know that the expected cost of covering $H$, call it $\COST\br{H}$ is at most $3\ln m \cdot \OPT$. By applying the Chernoff bound with $\delta=2$ we have that:
\begin{align*}
    \Pr[\COST\br{H} \geq 3\cdot \mathbb{E}[\COST\br{H}]] & 
    \leq \br{\frac{e^2}{3^3}}^{\mathbb{E}[\COST\br{H}]} \leq \e{-\mathbb{E}[\COST\br{H}]} \leq \frac{1}{m^3}.
\end{align*}
    \end{proof}

Combining the above lemmas with the $O\br{\log m}$-approximation for the covering problem we get that with high probability, the cost of the solution is bounded by $O\br{\log^2 m}\cdot \OPT$.
\end{proof}
